            left join ACA_COURSE c on c.COURSE_ID =a.COURSE_ID

            group by a.ASSIGNMENT_ID,a.ASSIGNMENT_TITLE,a.DESCRIPTION,a.SUBMISSION_DT,b.SUBMISSION_DT, b.ASSIGNMENT_FILE,b.STUDENT_ID, c.COURSE_TITLE
        ")->result();
